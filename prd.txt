1. Executive Summary
The Flower Species Identification System is a full-stack AI-powered web application that enables users to upload flower images and receive instant, accurate species identification using deep Convolutional Neural Networks (CNNs) trained from scratch and fine-tuned using transfer learning.

Project Vision: Democratize botanical knowledge by providing instant, accurate flower identification accessible to students, educators, gardeners, and nature enthusiasts — accessible from any browser, on any device.
Core Value Proposition: Transform flower identification from a time-consuming, expertise-dependent task into a 3-second automated process with 88%+ accuracy, supplemented by rich educational botanical content.
Unique Differentiator: Full end-to-end ownership of the ML pipeline — from raw dataset ingestion, custom data augmentation, model architecture selection, training, validation, optimization, and deployment — combined with a polished React frontend and FastAPI backend.

import scipy.io, shutil, os
from pathlib import Path

def organize(images_dir, labels_file, setid_file, output_dir):
    labels  = scipy.io.loadmat(labels_file)['labels'][0]  # shape (8189,)
    sets    = scipy.io.loadmat(setid_file)
    train_ids = sets['trnid'][0].tolist()
    val_ids   = sets['valid'][0].tolist()
    test_ids  = sets['tstid'][0].tolist()

    splits = {'train': train_ids, 'val': val_ids, 'test': test_ids}
    for split, ids in splits.items():
        for img_id in ids:
            cls = labels[img_id - 1]              # 1-indexed
            src = f'{images_dir}/image_{img_id:05d}.jpg'
            dst = f'{output_dir}/{split}/{cls:03d}/'
            Path(dst).mkdir(parents=True, exist_ok=True)
            shutil.copy(src, dst)

from tensorflow.keras.preprocessing.image import ImageDataGenerator
import cv2, numpy as np

TARGET_SIZE  = (224, 224)        # ResNet50 / EfficientNetB3 input
IMAGENET_MEAN = [0.485, 0.456, 0.406]
IMAGENET_STD  = [0.229, 0.224, 0.225]

def preprocess_image(img_path):
    img = cv2.imread(img_path)
    img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
    img = cv2.resize(img, TARGET_SIZE)
    img = img.astype(np.float32) / 255.0
    img = (img - IMAGENET_MEAN) / IMAGENET_STD   # ImageNet normalization
    return img

3.4 Model Architecture
Primary: ResNet50 with Custom Classification Head
import tensorflow as tf
from tensorflow.keras import layers, Model

def build_model(num_classes=102, base='resnet50'):
    # Load pre-trained backbone (ImageNet weights)
    if base == 'resnet50':
        backbone = tf.keras.applications.ResNet50(
            include_top=False,
            weights='imagenet',
            input_shape=(224, 224, 3)
        )
    elif base == 'efficientnetb3':
        backbone = tf.keras.applications.EfficientNetB3(
            include_top=False,
            weights='imagenet',
            input_shape=(300, 300, 3)   # EfficientNetB3 expects 300x300
        )

    # Stage 1: Freeze backbone
    backbone.trainable = False

    # Custom classification head
    x = backbone.output
    x = layers.GlobalAveragePooling2D()(x)
    x = layers.BatchNormalization()(x)
    x = layers.Dense(512, activation='relu')(x)
    x = layers.Dropout(0.5)(x)
    x = layers.Dense(256, activation='relu')(x)
    x = layers.Dropout(0.3)(x)
    outputs = layers.Dense(num_classes, activation='softmax')(x)

    model = Model(inputs=backbone.input, outputs=outputs)
    return model, backbone

Evaluation Script
from sklearn.metrics import classification_report, confusion_matrix
import numpy as np, mlflow

def evaluate_model(model, test_generator, class_names):
    y_pred_probs = model.predict(test_generator)
    y_pred       = np.argmax(y_pred_probs, axis=1)
    y_true       = test_generator.classes

    # Top-1 and Top-3 accuracy
    top1 = np.mean(y_pred == y_true)
    top3 = np.mean([y_true[i] in np.argsort(y_pred_probs[i])[-3:]
                    for i in range(len(y_true))])

    # Per-class report
    report = classification_report(y_true, y_pred,
                                   target_names=class_names, output_dict=True)
    cm = confusion_matrix(y_true, y_pred)

    # Log to MLflow
    with mlflow.start_run(run_name='final_evaluation'):
        mlflow.log_metric('top1_accuracy', top1)
        mlflow.log_metric('top3_accuracy', top3)
        mlflow.log_metric('macro_f1', report['macro avg']['f1-score'])
        mlflow.log_artifact('confusion_matrix.png')

    return {'top1': top1, 'top3': top3, 'report': report, 'cm': cm}

Confidence Calibration (Temperature Scaling)
# Temperature scaling post-training calibration
import scipy.optimize as opt

def temperature_scale(logits, T):
    return tf.nn.softmax(logits / T)

def calibrate(model, val_generator):
    logits = model.predict(val_generator, layer='pre_softmax')
    labels = val_generator.classes

    def nll(T):
        probs = temperature_scale(logits, T[0])
        return tf.keras.losses.sparse_categorical_crossentropy(labels, probs).numpy().mean()

    result = opt.minimize(nll, [1.5], bounds=[(0.1, 10.0)])
    optimal_T = result.x[0]
    print(f'Optimal Temperature: {optimal_T:.4f}')
    return optimal_T
# Target ECE < 0.10 after calibration

import tensorflow as tf, numpy as np
from functools import lru_cache

class FlowerClassifier:
    def __init__(self, model_path: str):
        self.model = tf.keras.models.load_model(model_path)
        self.class_names = self._load_class_names()
        self.confidence_threshold = 0.60

    @lru_cache(maxsize=100)
    def _load_class_names(self):
        with open('data/class_names.json') as f:
            return json.load(f)

    def predict(self, img_tensor, top_k=3):
        probs = self.model.predict(img_tensor, verbose=0)[0]  # (102,)
        top_indices = np.argsort(probs)[::-1][:top_k]

        results = []
        for rank, idx in enumerate(top_indices):
            results.append({
                'rank': rank + 1,
                'class_id': int(idx),
                'class_name': self.class_names[idx],
                'confidence': float(probs[idx]),
                'is_uncertain': probs[idx] < self.confidence_threshold
            })
        return results

6.4 Key Component — ImageUploader
// components/ImageUploader/index.tsx
import { useDropzone } from 'react-dropzone';
import { useCamera } from '@/hooks/useCamera';

export const ImageUploader = ({ onImageSelected }) => {
  const { getRootProps, getInputProps, isDragActive } = useDropzone({
    accept: { 'image/jpeg': [], 'image/png': [], 'image/webp': [] },
    maxSize: 10 * 1024 * 1024,   // 10 MB
    onDrop: (acceptedFiles) => onImageSelected(acceptedFiles[0]),
  });

  return (
    <div className='flex flex-col gap-4 items-center'>
      <div {...getRootProps()} className={`
        border-2 border-dashed rounded-2xl p-12 cursor-pointer
        transition-colors w-full text-center
        ${isDragActive ? 'border-green-500 bg-green-50' : 'border-gray-300 hover:border-green-400'}
      `}>
        <input {...getInputProps()} />
        <p className='text-gray-500'>Drag & drop or click to upload</p>
      </div>
      <CameraCapture onCapture={onImageSelected} />
    </div>
  );
};

6.5 Key Component — PredictionCard
// components/ResultsDisplay/PredictionCard.tsx
import { motion } from 'framer-motion';

export const PredictionCard = ({ prediction, rank }) => {
  const isTop = rank === 1;
  return (
    <motion.div
      initial={{ opacity: 0, y: 20 }}
      animate={{ opacity: 1, y: 0 }}
      transition={{ delay: rank * 0.15 }}
      className={`rounded-2xl p-6 shadow-md ${isTop ? 'bg-green-50 border-2 border-green-400' : 'bg-white border border-gray-200'}`}
    >
      <div className='flex items-center gap-4'>
        <img src={prediction.details.reference_image_url} className='w-20 h-20 rounded-xl object-cover' />
        <div className='flex-1'>
          <h2 className='text-xl font-bold text-gray-900'>{prediction.species.common_name}</h2>
          <p className='text-sm italic text-gray-500'>{prediction.species.scientific_name}</p>
          <ConfidenceBar value={prediction.confidence} />
        </div>
      </div>
      {isTop && <BotanicalInfo details={prediction.details} />}
      {prediction.is_uncertain && (
        <p className='text-yellow-600 text-sm mt-2'>⚠ Low confidence — result may be inaccurate</p>
      )}
    </motion.div>
  );
};

}

Appendix B: Glossary
ML / Technical Terms
Term
Definition
CNN
Convolutional Neural Network — deep learning architecture specialized for image analysis
Transfer Learning
Using pre-trained model weights (e.g., ImageNet) as starting point for a new task
Top-K Accuracy
Percentage of predictions where the correct answer appears in the top K results
Confidence Score
Model's probability estimate for a prediction (0–1 scale)
Data Augmentation
Artificially expanding training data by applying image transformations
Inference Latency
Time taken to generate a prediction from a single input image
F1 Score
Harmonic mean of precision and recall — balanced accuracy metric
ECE
Expected Calibration Error — measures how well confidence scores match true accuracy
Focal Loss
Loss function that down-weights easy examples, focusing training on hard cases
ONNX
Open Neural Network Exchange — cross-platform model format
MLflow
Open-source platform for ML experiment tracking and model registry
DVC
Data Version Control — Git-like versioning for datasets and models
Temperature Scaling
Post-training method to calibrate confidence scores using a single parameter T

Term
Definition
Species
Basic unit of biological classification (e.g., Rosa damascena)
Family
Taxonomic rank grouping related genera (e.g., Rosaceae for roses)
Common Name
Vernacular name used by the general public (e.g., 'Rose')
Scientific Name
Latinized binomial nomenclature (e.g., Helianthus annuus)
Distinguishing Feat
Visual characteristics that differentiate one species from another
